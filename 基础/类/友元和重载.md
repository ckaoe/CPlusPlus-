```cpp
#include<iostream>
using namespace std;

class Person
{
private:
    int m_a{};
public:
    Person(int a):m_a{a}{}
    // 友元关键词：friend和重载关键词operator+
    friend Person operator+(const Person &p1, const Person &p2);

    int getPerson() { return m_a; }

};

Person operator+(const Person &p1, const Person &p2)
{
    return p1.m_a + p2.m_a;
}



int main()
{
    Person p1{1};
    Person p2{2};
    Person person{p1 + p2};
    cout << person.getPerson() << endl;

    return 0;
}
```

```cpp
#include<iostream>
using namespace std;

class Cents
{
private:
    int m_cents{};
    int m_ceuts{};
public:
    Cents(int cents, int ceuts) :m_cents{ cents }, m_ceuts{ ceuts } {}
    friend Cents operator+(const Cents& c1, const Cents& c2);
    // 不同类型的友元函数重载，必须要有一个本类类型，否则会报错，
    friend Cents operator+(int c1, const Cents& c2);
    // 这条会报错。
    //friend Cents operator+(int c1, int c2);
    int getCents() { return m_cents; }
};

Cents operator+(const Cents& c1, const Cents& c2)
{
    return c1.m_cents + c2.m_cents;
}

Cents operator+(int c1, const Cents& c2)
{
    return c1 + c2.m_cents;
}

int main()
{
    Cents cents{ 4 };
    Cents cent{ 2 };

    Cents cen{ cents + cent };
    cout << cen.getCents() << "-----\n";

    Cents c1{ 4 };
    Cents cc{ 6 + c1 };
    cout << cc.getCents() << "------\n";


    return 0;
}
```

> 不包含友元的重载\

```cpp
#include <iostream>
using namespace std;

class Cents
{
private:
    int m_cents{};
public:
    Cents(int cents): m_cents{cents}{}
    int getCents() const { return m_cents; }
};

Cents operator+(const Cents& c1, const  Cents& c2)
{
    return Cents{c1.getCents() + c2.getCents()};
}

int main()
{
    Cents cen{1};
    Cents ce{2};
    Cents c{cen + ce};
    cout << c.getCents() << endl;

    return 0;
}
```
