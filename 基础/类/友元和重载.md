```cpp
#include<iostream>
using namespace std;

class Person
{
private:
    int m_a{};
public:
    Person(int a):m_a{a}{}
    // 友元关键词：friend和重载关键词operator+
    friend Person operator+(const Person &p1, const Person &p2);

    int getPerson() { return m_a; }

};

Person operator+(const Person &p1, const Person &p2)
{
    return p1.m_a + p2.m_a;
}



int main()
{
    Person p1{1};
    Person p2{2};
    Person person{p1 + p2};
    cout << person.getPerson() << endl;

    return 0;
}
```
