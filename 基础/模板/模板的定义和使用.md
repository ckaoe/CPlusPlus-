在传入参数的时候选择传入常量地址是一个非常好的选择。
```C++
template <typename T>
T minimum(const T& lhs, const T& rhs)
{
	return lhs < rhs ? lhs : rhs;
}

```

----

```cpp
template <typename T>
struct Pair
{
    T first{};
    T second{};
};

int main()
{
    Pair<int> p1{5, 6};  //显式实例化为int类型

    Pair<double> p2{2.2, 3.3}; // 显式初始化为double

    return 0;
}
```
----

```cpp
#include <iostream>
using namespace std;

template <typename T>
struct Pair
{
	int a{};
	int b{};
};

int max(Pair<int> p)
{
	return p.a < p.b ? p.b : p.a;
}
```
----

```cpp
#include <iostream>
using namespace std;

template <typename T, typename U>
struct Pair
{
	T first{};
	U second{};
};

int max(Pair<int, double> p)
{
	return p.first < p.second ? p.second : p.first;
}

```

```cpp
#include <iostream>
using namespace std;

template <typename T, typename U>
void print(pair<T, U> p)
{
	std::cout << '[' << p.first << ", " << p.second << ']';
}

int main()
{
	pair p{ 1,2 };
	print(p);

	return 0;
}
```
