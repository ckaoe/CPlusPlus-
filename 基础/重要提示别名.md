# 命名空间别名
`namespace a = std;`

```c++
#include<iostream>
// 创建命名空间的别名
namespace a = std;

int main()
{
    a::cout << "hello world\n";
    
    return 0;
}
```


# 类型别名
`using b = int;`

```c++
#include<iostream>
namespace a = std;
using b = int;

int main()
{
    b c = 0;
    a::cout << c << a::endl;
    a::cout << "hello world\n";
    
    return 0;
}
```
